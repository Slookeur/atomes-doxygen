101void fill_bond_model_row (int p, int a, int b, GtkTreeStore * store)
102{
103 GtkTreeIter bond_level;
104 gchar * str_a, * str_b, * str_c;
105 struct project * this_proj = get_project_by_id(p);
106 struct atom * at, * bt;
107 gboolean pbc = this_proj -> cell.pbc;
108 struct distance dist;
109 int s = this_proj -> modelgl -> anim -> last -> img -> step;
110 at = & this_proj -> atoms[s][a];
111 bt = & this_proj -> atoms[s][b];
112 bond_id ++;
113 dist = distance_3d (& this_proj -> cell, (this_proj -> cell.npt) ? s : 0, at, bt);
114 str_a = g_strdup_printf("%s<sub>%d</sub>", this_proj -> chemistry -> label[at -> sp], a+1);
115 str_b = g_strdup_printf("%s<sub>%d</sub>", this_proj -> chemistry -> label[bt -> sp], b+1);
116 str_c = g_strdup_printf("%.3lf", dist.length);
117 if (pbc)
118 {
119 gtk_tree_store_append (store, & bond_level, NULL);
120 gtk_tree_store_set (store, & bond_level, 0, bond_id,
121 1, str_a,
122 2, str_b,
123 3, str_c,
124 4, dist.pbc, -1);
125 }
126 else
127 {
128 gtk_tree_store_append (store, & bond_level, NULL);
129 gtk_tree_store_set (store, & bond_level, 0, bond_id,
130 1, str_a,
131 2, str_b,
132 3, str_c, -1);
133 }
134 g_free (str_a);
135 g_free (str_b);
136 g_free (str_c);
137}

285void measure_set_color (GtkTreeViewColumn * col, GtkCellRenderer * renderer, GtkTreeModel * mod, GtkTreeIter * iter, gpointer data)
286{
287 int i, j;
288 tint * dat = (tint *) data;
289 gboolean docolor = FALSE;
290 gtk_tree_model_get (mod, iter, 0, & i, -1);
291 struct project * this_proj = get_project_by_id(dat -> a);
292 image * img = this_proj -> modelgl -> anim -> last -> img;
293 int pi = get_selection_type (this_proj -> modelgl);
294 if (img -> selected[pi] -> selected)
295 {
296 if ((dat -> c == 2 && img -> selected[pi] -> selected < MAX_IN_SELECTION-10) || (dat -> c < 2 && img -> selected[pi] -> selected < MAX_IN_SELECTION))
297 {
298 switch (dat -> c)
299 {
300 case 0:
301 j = num_bonds (img -> selected[pi] -> selected);
302 if (img -> selected[pi] -> selected_bonds[i]) docolor = TRUE;
303 break;
304 case 1:
305 j = num_angles (img -> selected[pi] -> selected);
306 if (img -> selected[pi] -> selected_angles[i]) docolor = TRUE;
307 break;
308 case 2:
309 j = num_dihedrals (img -> selected[pi] -> selected);
310 if (img -> selected[pi] -> selected_dihedrals[i]) docolor = TRUE;
311 break;
312 }
313 }
314 }
315 set_renderer_color (docolor, renderer, init_color (i, j));
316}

392G_MODULE_EXPORT void measure_tree_button_event (GtkWidget * widget, double event_x, double event_y, guint event_button, gpointer data)
393{
394 if (event_button == 1)
395 {
396 tint * dat = (tint *)data;
397 glwin * view = get_project_by_id(dat -> a) -> modelgl;
398 int pi = get_selection_type (view);
399 GtkTreeModel * measure_model = gtk_tree_view_get_model(GTK_TREE_VIEW(widget));
400 GtkTreePath * path;
401 GtkTreeViewColumn * column;
402 int i, j;
403#ifdef GTK4
404 int e_x, e_y;
405 gtk_tree_view_convert_widget_to_bin_window_coords (GTK_TREE_VIEW(widget), event_x, event_y, & e_x, & e_y);
406 if (gtk_tree_view_get_path_at_pos (GTK_TREE_VIEW(widget), e_x, e_y, & path, & column, & i, & j))
407#else
408 if (gtk_tree_view_get_path_at_pos (GTK_TREE_VIEW(widget), event_x, event_y, & path, & column, & i, & j))
409#endif
410 {
411 GtkTreeIter row;
412 if (gtk_tree_model_get_iter (measure_model, & row, path))
413 {
414 gtk_tree_model_get (measure_model, & row, 0, & j, -1);
415 // select bonds, angles or dihedraks:
416 switch (dat -> c)
417 {
418 case 0:
419 view -> anim -> last -> img -> selected[pi] -> selected_bonds[j] = ! view -> anim -> last -> img -> selected[pi] -> selected_bonds[j];
420 break;
421 case 1:
422 view -> anim -> last -> img -> selected[pi] -> selected_angles[j] = ! view -> anim -> last -> img -> selected[pi] -> selected_angles[j];
423 break;
424 case 2:
425 view -> anim -> last -> img -> selected[pi] -> selected_dihedrals[j] = ! view -> anim -> last -> img -> selected[pi] -> selected_dihedrals[j];
426 break;
427 }
428 }
429 }
430 view -> create_shaders[MEASU] = TRUE;
431 update (view);
432 }
433}

481GtkWidget * create_selection_tree (glwin * view, int sid, int mid)
482{
483 int i, j;
484 GtkTreeViewColumn * sel_col[7];
485 GtkCellRenderer * sel_cell[7];
486 GtkTreeStore * sel_model;
487 gchar * ctitle[3][7]={{"Id", "Atom 1", "Atom 2", "Distance []", "Using PBC", "NULL", "NULL"},
488 {"Id", "Atom 1", "Atom 2", "Atom 3", " []", "Using PBC", "NULL"},
489 {"Id", "Atom 1", "Atom 2", "Atom 3", "Atom 4", " []", "Using PBC"}};
490 gchar * ctype[3][7]={{"text", "text", "text", "text", "active", "text", "text"},
491 {"text", "text", "text", "text", "text", "active", "text"},
492 {"text", "text", "text", "text", "text", "text", "active"}};
493 int tree_dim[3]={4, 5, 6};
494 GType col_type[3][7]= {{G_TYPE_INT, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_BOOLEAN, G_TYPE_STRING, G_TYPE_STRING},
495 {G_TYPE_INT, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_BOOLEAN, G_TYPE_STRING},
496 {G_TYPE_INT, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_BOOLEAN}};
497 if (get_project_by_id(view -> proj) -> cell.pbc)
498 {
499 sel_model = gtk_tree_store_newv (tree_dim[mid]+1, col_type[mid]);
500 j = 1;
501 }
502 else
503 {
504 sel_model = gtk_tree_store_newv (tree_dim[mid], col_type[mid]);
505 j = 0;
506 }
507
508 GtkWidget * selection_tree = gtk_tree_view_new_with_model(GTK_TREE_MODEL(sel_model));
509 for (i=0; i<4+j+mid; i++)
510 {
511 if ((mid == 0 && i < 4) || (mid == 1 && i < 5) || (mid == 2 && i < 6))
512 {
513 sel_cell[i] = gtk_cell_renderer_text_new();
514 sel_col[i] = gtk_tree_view_column_new_with_attributes (ctitle[mid][i], sel_cell[i], ctype[mid][i], i, NULL);
515 if (i == 0) gtk_tree_view_column_set_visible (sel_col[i], FALSE);
516 }
517 else if ((mid == 0 && i == 4) || (mid == 1 && i == 5) || (mid == 2 && i == 6))
518 {
519 sel_cell[i] = gtk_cell_renderer_toggle_new ();
520 sel_col[i] = gtk_tree_view_column_new_with_attributes (ctitle[mid][i], sel_cell[i], ctype[mid][i], i, NULL);
521 }
522 gtk_tree_view_column_set_alignment (sel_col[i], 0.5);
523 gtk_tree_view_append_column(GTK_TREE_VIEW(selection_tree), sel_col[i]);
524 if (i > 0 && i < 4+mid)
525 {
526 if (mid < 2)
527 {
528 gtk_tree_view_column_set_cell_data_func (sel_col[i], sel_cell[i], measure_set_color_and_markup, & view -> colorp[i][mid], NULL);
529 }
530 else
531 {
532 gtk_tree_view_column_set_cell_data_func (sel_col[i], sel_cell[i], dihedral_set_color_and_markup, & view -> colorp[i][0], NULL);
533 }
534 }
535 else if (i > 0)
536 {
537 gtk_tree_view_column_set_cell_data_func (sel_col[i], sel_cell[i], measure_set_visible, & view -> colorp[i][0], NULL);
538 }
539 }
540 image * img = view -> anim -> last -> img;
541 if (img -> selected[sid] -> selected < MAX_IN_SELECTION)
542 {
543 if ( (mid == 0 && img -> selected[sid] -> selected > 1)
544 || (mid == 1 && img -> selected[sid] -> selected > 2)
545 || (mid == 2 && img -> selected[sid] -> selected > 3 && img -> selected[sid] -> selected < MAX_IN_SELECTION-10))
546 {
547 if (mid == 0)
548 {
549 bond_id = -1;
550 bonds_loop (view, 2, sid, GTK_TREE_STORE(sel_model));
551#ifdef GTK4
552 add_widget_gesture_and_key_action (selection_tree, "bond-tree-pressed", G_CALLBACK(measure_tree_button_pressed), & view -> colorp[0][mid],
553 NULL, NULL, NULL, NULL, NULL, NULL,
554 NULL, NULL, NULL, NULL, NULL, NULL);
555#else
556 g_signal_connect (G_OBJECT(selection_tree), "button_press_event", G_CALLBACK(measure_tree_selection_event), & view -> colorp[0][mid]);
557#endif
558 }
559 else if (mid == 1)
560 {
561 angle_id = -1;
562 angles_loop (view, 2, sid, GTK_TREE_STORE(sel_model));
563#ifdef GTK4
564 add_widget_gesture_and_key_action (selection_tree, "angle-tree-pressed", G_CALLBACK(measure_tree_button_pressed), & view -> colorp[0][mid],
565 NULL, NULL, NULL, NULL, NULL, NULL,
566 NULL, NULL, NULL, NULL, NULL, NULL);
567#else
568 g_signal_connect (G_OBJECT(selection_tree), "button_press_event", G_CALLBACK(measure_tree_selection_event), & view -> colorp[0][mid]);
569#endif
570 }
571 else if (mid == 2)
572 {
573 dihedral_id = -1;
574 dihedrals_loop (view, 2, sid, GTK_TREE_STORE(sel_model));
575/*#ifdef GTK4
576 add_widget_gesture_and_key_action (selection_tree, "dihedral-tree-pressed", G_CALLBACK(measure_tree_button_pressed), & view -> colorp[0][mid],
577 NULL, NULL, NULL, NULL, NULL, NULL,
578 NULL, NULL, NULL, NULL, NULL, NULL);
579#else
580 g_signal_connect (G_OBJECT(selection_tree), "button_press_event", G_CALLBACK(measure_tree_selection_event), & view -> colorp[0][mid]);
581#endif*/
582 }
583 }
584 }
585 GtkTreeSelection * selection = gtk_tree_view_get_selection(GTK_TREE_VIEW(selection_tree));
586 gtk_tree_selection_set_mode (selection, GTK_SELECTION_NONE);
587 gtk_tree_view_expand_all (GTK_TREE_VIEW(selection_tree));
588 return selection_tree;
589}
