219void clear_field_atoms (struct field_molecule * fmol, struct field_atom * at, int mols, int * mol)
220{
221 int i, j, k, l;
222 l = 0;
223 for (i=0; i < at -> num; i++)
224 {
225 k = at -> list[i];
226 for (j=0; j < mols; j++)
227 {
228 if (tmp_proj -> atoms[0][k].coord[2] == mol[j])
229 {
230 l ++;
231 }
232 }
233 }
234 int * r_list = allocint (l);
235 int * s_list = allocint (l);
236 int * t_list = allocint (l);
237 l = -1;
238 at -> frozen = 0;
239 for (i=0; i < at -> num; i++)
240 {
241 k = at -> list[i];
242 for (j=0; j < mols; j++)
243 {
244 if (tmp_proj -> atoms[0][k].coord[2] == mol[j])
245 {
246 l ++;
247 r_list[l] = k;
248 s_list[l] = at -> list_id[i];
249 t_list[l] = at -> frozen_id[i];
250 if (t_list[l]) at -> frozen ++;
251 }
252 }
253 }
254 at -> num = l+1;
255 at -> list = NULL;
256 at -> list = allocint (at -> num);
257 at -> list_id = NULL;
258 at -> list_id = allocint (at -> num);
259 at -> frozen_id = NULL;
260 at -> frozen_id = allocbool (at -> num);
261 for (i=0; i < at -> num; i++)
262 {
263 at -> list[i] = r_list[i];
264 at -> list_id[i] = s_list[i];
265 at -> frozen_id[i] = t_list[i];
266 j = at -> list[i];
267 k = at -> list_id[i];
268 for (l=0; l<mols; l++)
269 {
270 if (tmp_proj -> atoms[0][j].coord[2] == mol[l])
271 {
272 atomd_id_save[k][l].a = at -> id;
273 atomd_id_save[k][l].b = i;
274 }
275 }
276 }
277 g_free (r_list);
278 g_free (s_list);
279 g_free (t_list);
280}

310G_MODULE_EXPORT void run_add_molecule_to_field (GtkDialog * dialog, gint response_id, gpointer data)
311{
312 int i, j, k;
313 int * old_mol = NULL;
314 gchar * str;
315 gboolean done = FALSE;
316 gboolean saveit = FALSE;
317 // Id - Name - Multiplicity - Select
318 // While selected < m apply inactif
319 switch (response_id)
320 {
321 case GTK_RESPONSE_APPLY:
322 if (a_mol > 0 && a_mol < tmp_fmol -> multi)
323 {
324 if (a_mol > 1)
325 {
326 str = g_strdup_printf ("Fragments N%d", new_mol[0]+1);
327 if (a_mol > 2)
328 {
329 for (i=1; i<a_mol-1; i++)
330 {
331 str = g_strdup_printf ("%s, %d", str, new_mol[i]+1);
332 }
333 }
334 str = g_strdup_printf ("%s and %d have been selected !", str, new_mol[a_mol-1]+1);
335 str = g_strdup_printf ("%s\nConfirm this choice and create a new field molecule to describe these fragments ?", str);
336 }
337 else
338 {
339 str = g_strdup_printf ("Fragment N%d has been selected !", new_mol[0]+1);
340 str = g_strdup_printf ("%s\nConfirm this choice and create a new field molecule to describe this fragment ?", str);
341 }
342 selection_confirmed = FALSE;
343 field_question (str, G_CALLBACK(confirm_selection), NULL);
344 g_free (str);
345 if (selection_confirmed)
346 {
347 done = TRUE;
348 struct field_molecule * old_fmol = get_active_field_molecule(row_id);
349 struct field_molecule * next_fmol = duplicate_field_molecule (old_fmol);
350 get_active_field_molecule(tmp_field -> molecules-1) -> next = next_fmol;
351 next_fmol -> prev = get_active_field_molecule(tmp_field -> molecules-1);
352 next_fmol -> id = tmp_field -> molecules;
353 next_fmol -> multi = a_mol;
354 next_fmol -> fragments = NULL;
355 next_fmol -> fragments = allocint(a_mol);
356 for (i=0; i<a_mol; i++) next_fmol -> fragments[i] = new_mol[i];
357 // Atoms_id and field atoms
358 atomd_id_save = g_malloc (next_fmol-> mol -> natoms*sizeof*atomd_id_save);
359 for (i=0; i<next_fmol-> mol -> natoms; i++)
360 {
361 atomd_id_save[i] = g_malloc (next_fmol -> multi*sizeof*atomd_id_save[i]);
362 }
363 tmp_fat = next_fmol -> first_atom;
364 for (i=0; i < next_fmol -> atoms; i++)
365 {
366 clear_field_atoms (next_fmol, tmp_fat, a_mol, new_mol);
367 tmp_fat -> name = g_strdup_printf ("%s", set_field_atom_name (tmp_fat, next_fmol));
368 if (tmp_fat -> next != NULL) tmp_fat = tmp_fat -> next;
369 }
370 g_free (next_fmol -> atoms_id);
371 next_fmol -> atoms_id = NULL;
372 next_fmol -> atoms_id = g_malloc (next_fmol-> mol -> natoms*sizeof*next_fmol -> atoms_id);
373 for (i=0; i<next_fmol-> mol -> natoms; i++)
374 {
375 next_fmol -> atoms_id[i] = g_malloc (next_fmol -> multi*sizeof*next_fmol -> atoms_id[i]);
376 for (j=0; j<next_fmol -> multi; j++)
377 {
378 next_fmol -> atoms_id[i][j].a = atomd_id_save[i][j].a;
379 next_fmol -> atoms_id[i][j].b = atomd_id_save[i][j].b;
380 }
381 }
382 g_free (atomd_id_save);
383 old_mol = allocint(old_fmol -> multi - a_mol);
384 k = -1;
385 for (i=0; i<old_fmol -> multi; i++)
386 {
387 saveit = TRUE;
388 for (j=0; j<a_mol; j++)
389 {
390 if (new_mol[j] == old_fmol -> fragments[i])
391 {
392 saveit = FALSE;
393 break;
394 }
395 }
396 if (saveit)
397 {
398 k++;
399 old_mol[k] = old_fmol -> fragments[i];
400 }
401 }
402 new_mol = NULL;
403 // Now we deal with atoms_id and the field atoms
404 atomd_id_save = g_malloc (old_fmol-> mol -> natoms*sizeof*atomd_id_save);
405 for (i=0; i<old_fmol -> mol -> natoms; i++)
406 {
407 atomd_id_save[i] = g_malloc ((old_fmol -> multi - a_mol)*sizeof*atomd_id_save[i]);
408 }
409 tmp_fat = old_fmol -> first_atom;
410 for (i=0; i < old_fmol -> atoms; i++)
411 {
412 clear_field_atoms (old_fmol, tmp_fat, old_fmol -> multi - a_mol, old_mol);
413 if (tmp_fat -> next != NULL) tmp_fat = tmp_fat -> next;
414 }
415 old_fmol -> multi -= a_mol;
416 g_free (old_fmol -> atoms_id);
417 old_fmol -> atoms_id = NULL;
418 old_fmol -> atoms_id = g_malloc (old_fmol-> mol -> natoms*sizeof*old_fmol -> atoms_id);
419 for (i=0; i<old_fmol-> mol -> natoms; i++)
420 {
421 old_fmol -> atoms_id[i] = g_malloc (old_fmol -> multi*sizeof*old_fmol -> atoms_id[i]);
422 for (j=0; j<old_fmol -> multi; j++)
423 {
424 old_fmol -> atoms_id[i][j].a = atomd_id_save[i][j].a;
425 old_fmol -> atoms_id[i][j].b = atomd_id_save[i][j].b;
426 }
427 }
428 g_free (atomd_id_save);
429 old_fmol -> fragments = NULL;
430 old_fmol -> fragments = allocint(old_fmol -> multi);
431 for (i=0; i<old_fmol -> multi; i++) old_fmol -> fragments[i] = old_mol[i];
432 old_mol = NULL;
433 row_id = tmp_field -> molecules;
434 tmp_field -> molecules ++;
435 edit_field_prop (NULL, NULL, GINT_TO_POINTER(0));
436 clean_up_molecules_info (TRUE);
437 }
438 }
439 break;
440 default:
441 field_unselect_all ();
442 done = TRUE;
443 break;
444 }
445 if (done)
446 {
447 g_free (new_mol);
448 new_mol = NULL;
449 destroy_this_dialog (dialog);
450 }
451}

529void merge_all_atoms_to_mol (struct field_molecule * new_mol, int mstart)
530{
531 struct field_atom * fat, * fbt;
532 fat = get_active_atom (new_mol -> id, new_mol -> atoms-1);
533 int i, j, k, l, m, n;
534 for (i=0; i<fat -> num; i++)
535 {
536 j = fat -> list[i];
537 k = fat -> list_id[i];
538 l = new_mol -> atoms_id[k][0].a;
539 fbt = get_active_atom (new_mol -> id, l);
540 fbt -> list[fbt -> num] = j;
541 fbt -> list_id[fbt -> num] = k;
542 m = tmp_proj -> atoms[0][j].coord[2];
543 for (n=mstart; n<new_mol -> multi; n++)
544 {
545 if (new_mol -> fragments[n] == m)
546 {
547 new_mol -> atoms_id[k][n].a = fbt -> id;
548 new_mol -> atoms_id[k][n].b = fbt -> num;
549 break;
550 }
551 }
552 fbt -> num ++;
553 }

565void prepare_atoms_to_merge (struct field_atom * at, struct field_molecule * new_mol, struct field_molecule * old_mol)
566{
567 int i, j;
568 int * saved_list = allocint (at -> num);
569 int * saved_list_id = allocint (at -> num);
570 int * saved_frozen_id = allocbool (at -> num);
571 for (j=0; j < at -> num; j++)
572 {
573 saved_list[j] = at -> list[j];
574 saved_list_id[j] = at -> list_id[j];
575 saved_frozen_id[j] = at -> frozen_id[j];
576 }
577 i = at -> num;
578 i /= (new_mol -> multi - old_mol -> multi);
579 i *= old_mol -> multi;
580 g_free (at -> list);
581 at -> list = allocint (at -> num + i);
582 g_free (at -> list_id);
583 at -> list_id = allocint (at -> num + i);
584 g_free (at -> frozen_id);
585 at -> frozen_id = allocbool (at -> num + i);
586 for (j=0; j< at -> num; j++)
587 {
588 at -> list[j] = saved_list[j];
589 at -> list_id[j] = saved_list_id[j];
590 at -> frozen_id[j] = saved_frozen_id[j];
591 }
592 g_free (saved_list);
593 g_free (saved_list_id);
594 g_free (saved_frozen_id);
595}

606{
607 struct field_atom * fat, * fbt;

613 fat -> list_id = allocint (fmol -> mol -> natoms * fmol -> multi);
614 fbt = fmol -> first_atom;

616 i = 0;
617 while (fbt != NULL)
618 {
619 for (j=0; j<fbt -> num; j++)
620 {
621 fat -> list[i] = fbt -> list[j];
622 fat -> list_id[i] = fbt -> list_id[j];
623 i ++;
624 }
625 fbt = fbt -> next;
626 }
627 return fat;

639G_MODULE_EXPORT void run_remove_molecule_from_field (GtkDialog * rmol, gint response_id, gpointer data)
640{
641 int i, j;
642 gboolean done = FALSE;
643 switch (response_id)
644 {
645 case GTK_RESPONSE_APPLY:
646 if (a_mol)
647 {
648 done = TRUE;
649 struct field_molecule * to_merge = get_active_field_molecule(new_mol[0]);
650 int * old_mol = allocint (to_merge -> multi);
651 for (i=0; i<to_merge -> multi; i++) old_mol[i] = to_merge -> fragments[i];
652 g_free (to_merge -> fragments);
653 to_merge -> fragments = allocint (to_merge -> multi + to_remove -> multi);
654 for (i=0; i<to_merge -> multi; i++) to_merge -> fragments[i] = old_mol[i];
655 old_mol = NULL;
656 for (i=to_merge -> multi; i < to_merge -> multi + to_remove -> multi; i++)
657 {
658 to_merge -> fragments[i] = to_remove -> fragments[i - to_merge -> multi];
659 }
660 atomd_id_save = g_malloc (to_merge -> mol -> natoms*sizeof*atomd_id_save);
661 for (i=0; i<to_merge -> mol -> natoms; i++)
662 {
663 atomd_id_save[i] = g_malloc ((to_merge -> multi+to_remove -> multi)*sizeof*atomd_id_save[i]);
664 for (j=0; j<to_merge -> multi; j++)
665 {
666 atomd_id_save[i][j].a = to_merge -> atoms_id[i][j].a;
667 atomd_id_save[i][j].b = to_merge -> atoms_id[i][j].b;
668 }
669 for (j=to_merge -> multi; j<to_merge -> multi+to_remove -> multi; j++)
670 {
671 atomd_id_save[i][j].a = -1;
672 atomd_id_save[i][j].b = -1;
673 }
674 }
675 to_merge -> multi += to_remove -> multi;
676 g_free (to_merge -> atoms_id);
677 to_merge -> atoms_id = g_malloc (to_merge -> mol -> natoms*sizeof*to_merge -> atoms_id);
