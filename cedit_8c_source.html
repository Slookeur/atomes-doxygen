92void prepbox (int k, int l, int m)
93{
94 int i, n, o, p;
95 gchar * str;
96
97 struct project * extra_proj;
98 struct project * this_proj = get_project_by_id(k);
99 str = g_strdup_printf ("%s - %s", prepare_for_title(this_proj -> name), this_proj -> curves[l][m] -> name);
100 combo_text_append (setcolorbox, str);
101 g_free (str);
102 struct cextra * ctmp = this_proj -> curves[l][m] -> extrac -> first;
103 for ( i=0 ; i < this_proj -> curves[l][m] -> extrac -> extras ; i++ )
104 {
105 n = ctmp -> id.a;
106 o = ctmp -> id.b;
107 p = ctmp -> id.c;
108 extra_proj = get_project_by_id(n);
109 str = g_strdup_printf ("%s - %s", prepare_for_title(extra_proj -> name), extra_proj -> curves[o][p] -> name);
110 combo_text_append (setcolorbox, str);
111 g_free (str);
112 if (ctmp -> next != NULL) ctmp = ctmp -> next;
113 }
114 if (this_proj -> curves[l][m] -> extrac -> extras > 0)
115 {
116 widget_set_sensitive (setcolorbox, 1);
117 }
118 else
119 {
120 widget_set_sensitive (setcolorbox, 0);
121 }
122 gtk_widget_set_size_request (setcolorbox, -1, 40);
123 gtk_combo_box_set_active (GTK_COMBO_BOX (setcolorbox), 0);
124 g_signal_connect (G_OBJECT(setcolorbox), "changed", G_CALLBACK(choose_set), NULL);
125}

159static void fill_proj_model (GtkTreeStore * store)
160{
161 GtkTreeIter projlevel;
162 GtkTreeIter calclevel;
163 GtkTreeIter curvelevel;
164 struct project * this_proj;
165 int i, j, k;
166 int start, end, step;
167 gboolean append;
168
169 /* Append a top level row and leave it empty */
170 if (ppath != NULL) g_free (ppath);
171 if (cpath != NULL) g_free (cpath);
172 ppath = g_malloc0 (nprojects*sizeof*ppath);
173 cpath = g_malloc0 (nprojects*sizeof*cpath);
174 for (i=0; i<nprojects; i++)
175 {
176 this_proj = get_project_by_id(i);
177 gtk_tree_store_append (store, & projlevel, NULL);
178 gtk_tree_store_set (store, & projlevel, 0, 0, 1, prepare_for_title(this_proj -> name), 2, TRUE, 3, -1, -1);
179 ppath[i] = gtk_tree_model_get_path ((GtkTreeModel *)store, & projlevel);
180 if (activer == 0 || activer == 3)
181 {
182 start = 0;
183 end = 4;
184 step = 3;
185 }
186 else if (activer == 1 || activer == 2)
187 {
188 start = 1;
189 end = 3;
190 step = 1;
191 }
192 else
193 {
194 start = activer;
195 end = start + 1;
196 step = 1;
197 }
198 for (j=start; j<end; j=j+step)
199 {
200 if (this_proj -> initok[j])
201 {
202 gtk_tree_store_append (store, & calclevel, & projlevel);
203 gtk_tree_store_set (store, & calclevel, 0, 0, 1, graph_name[j], 2, TRUE, 3, -1, -1);
204 if (j == start)
205 {
206 cpath[i] = gtk_tree_model_get_path ((GtkTreeModel *)store, & calclevel);
207 }
208 for (k = 0 ; k < this_proj -> numc[j] ; k++)
209 {
210 append = FALSE;
211 if (i != activeg && this_proj -> curves[j][k] -> ndata != 0)
212 {
213 append = TRUE;
214 }
215 else if (((i != activeg) || (j != activer || k != activec)) && this_proj -> curves[j][k] -> ndata != 0)
216 {
217 append = TRUE;
218 }
219 if (append)
220 {
221 gtk_tree_store_append (store, & curvelevel, & calclevel);
222 gtk_tree_store_set (store, & curvelevel,
223 0, 1,
224 1, this_proj -> curves[j][k] -> name,
225 2, ! was_not_added (active_project -> curves[activer][activec] -> extrac, i, j, k),
226 3, i,
227 4, j,
228 5, k, -1);
229 }
230 }
231 }
232 }
233 }

282GtkWidget * create_projects_tree ()
283{
284 int i;
285 GtkTreeViewColumn * projcol[6];
286 GtkCellRenderer * projcell[6];
287 gchar * col_title[6] = {" ", "Data sets", "Select", " ", " ", " "};
288 gchar * ctype[6]={"text", "text", "active", "text", "text", "text"};
289 GType col_type[6] = {G_TYPE_INT, G_TYPE_STRING, G_TYPE_BOOLEAN, G_TYPE_INT, G_TYPE_INT, G_TYPE_INT};
290 projmodel = gtk_tree_store_newv (6, col_type);
291 projtree = gtk_tree_view_new_with_model(GTK_TREE_MODEL(projmodel));
292 for (i=0; i<6; i++)
293 {
294 if (i == 2)
295 {
296 projcell[i] = gtk_cell_renderer_toggle_new ();
297 projcol[i] = gtk_tree_view_column_new_with_attributes(col_title[i], projcell[i], ctype[i], i, NULL);
298 g_signal_connect (G_OBJECT(projcell[i]), "toggled", G_CALLBACK(toggle_curve), NULL);
299 gtk_tree_view_column_set_cell_data_func (projcol[i], projcell[i], set_visible_curve_data, NULL, NULL);
300 }
301 else
302 {
303 projcell[i] = gtk_cell_renderer_text_new();
304 projcol[i] = gtk_tree_view_column_new_with_attributes(col_title[i], projcell[i], ctype[i], i, NULL);
305 }
306 gtk_tree_view_append_column(GTK_TREE_VIEW(projtree), projcol[i]);
307 if (i == 0 || i > 2) gtk_tree_view_column_set_visible (projcol[i], FALSE);
308 }
309 fill_proj_model (projmodel);
310 g_object_unref (projmodel);
311 gtk_tree_view_expand_all (GTK_TREE_VIEW(projtree));
312 return projtree;
313}
