196G_MODULE_EXPORT void change_axis_pos_radio (GSimpleAction * action, GVariant * parameter, gpointer data)
197{
198 /* TOP_RIGHT = 0,
199 TOP_LEFT = 1,
200 BOTTOM_RIGHT = 2,
201 BOTTOM_LEFT = 3,
202 CENTER = 4 */
203 tint * the_data = (tint *)data;
204 glwin * view = get_project_by_id(the_data -> a) -> modelgl;
205 const gchar * pos = g_variant_get_string (parameter, NULL);
206 int lgt = strlen (pos);
207 gchar * name = g_strdup_printf ("%c%c", pos[lgt-2], pos[lgt-1]);
208 if (g_strcmp0(name, ".1") == 0)
209 {
210 g_free (name);
211 name = g_strdup_printf ("%.*s.0", lgt-2, pos);
212 g_action_group_activate_action ((GActionGroup *)view -> action_group, "set-axis-pos", g_variant_new_string((const gchar *)name));
213 g_free (name);
214 }
215 else
216 {
217 int i;
218 gchar * pos_name = NULL;
219 for (i=0; i<5; i++)
220 {
221 pos_name = g_strdup_printf ("set-axis-pos.%d.0", i);
222 if (g_strcmp0(pos, (const gchar *)pos_name) == 0)
223 {
224 view -> anim -> last -> img -> axispos = i;
225 g_free (pos_name);
226 pos_name = NULL;
227 break;
228 }
229 g_free (pos_name);
230 pos_name = NULL;
231 }
232 g_action_change_state (G_ACTION (action), parameter);

235 g_action_change_state (G_ACTION (action), parameter);
