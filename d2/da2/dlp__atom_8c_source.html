170void clean_old_atom (struct field_atom * at, int atos, int * atid)
171{
172 int h, i, j, k, l, m;
173 gboolean save;
174 h = at -> num - atos*tmp_fmol -> multi;
175 int * tlist = allocint (h);
176 int * dlist = allocint (h);
177 j = -1;
178 for (i=0; i < at -> num; i++)
179 {
180 save = TRUE;
181 for (k=0; k<atos; k++)
182 {
183 if (at -> list_id[i] == atid[k])
184 {
185 save = FALSE;
186 break;
187 }
188 }
189 if (save)
190 {
191 j ++;
192 tlist[j] = at -> list[i];
193 dlist[j] = at -> list_id[i];
194 }
195 }
196 at -> num = h;
197 g_free (at -> list);
198 at -> list = duplicate_int (h, tlist);
199 g_free (at -> list_id);
200 at -> list_id = duplicate_int (h, dlist);
201 for (i=0; i<h; i++)
202 {
203 k = at -> list[i];
204 l = at -> list_id[i];
205 for (m=0; m< tmp_fmol -> multi; m++)
206 {
207 if (tmp_proj -> atoms[0][k].coord[2] == tmp_fmol -> fragments[m]) break;
208 }
209 tmp_fmol -> atoms_id[l][m].b = i;
210 }
211 g_free (tlist);
212 g_free (dlist);

225void adjust_field_struct (int oid, int k, struct field_struct * olds)
226{
227 int i, j;
228 gboolean doit;
229 struct field_struct * str;
230 i = struct_id(k);
231 int * aids = allocint (i);
232
233 while (olds)
234 {
235 doit = TRUE;
236 if (oid < 0)
237 {
238 for (j=0; j<i; j++) aids[j] = olds -> aid[j];
239 }
240 else
241 {
242 for (j=0; j<i; j++)
243 {
244 if (olds -> aid[j] < oid)
245 {
246 aids[j] = olds -> aid[j];
247 }
248 else if (olds -> aid[j] > oid)
249 {
250 aids[j] = olds -> aid[j] - 1;
251 }
252 else
253 {
254 doit = FALSE;
255 }
256 }
257 }
258 if (doit)
259 {
260 j = get_struct_id_from_atom_id (k, aids);
261 if (j > 0)
262 {
263 str = get_active_struct (k, tmp_fmol -> id, j-1);
264 str -> def = duplicate_field_prop (olds -> def, k);
265 if (olds -> other)
266 {
267 duplicate_other_prop (oid, olds, str);
268 }
269 }
270 }
271 olds = olds -> next;
272 }
273 // if (k == 0) print_all_field_struct (tmp_fmol, k);
274 g_free (aids);

568void merging_atoms (struct field_atom * to_merge, struct field_atom * to_remove, gboolean upda)
569{
570 int * tmp_ato = allocint (to_merge -> num + to_remove -> num);
571 int * tmp_atd = allocint (to_merge -> num + to_remove -> num);
572 int * tmp_fre = allocint (to_merge -> num + to_remove -> num);
573 int i, j, k, l;
574 i = to_merge -> num + to_remove -> num;
575 for (j=0; j<to_merge -> num; j++)
576 {
577 tmp_ato[j] = to_merge -> list[j];
578 tmp_atd[j] = to_merge -> list_id[j];
579 tmp_fre[j] = to_merge -> frozen_id[j];
580 }
581
582 for (j=to_merge -> num; j<i ; j++)
583 {
584 tmp_ato[j] = to_remove -> list[j - to_merge -> num];
585 tmp_atd[j] = to_remove -> list_id[j - to_merge -> num];
586 tmp_fre[j] = to_remove -> frozen_id[j - to_merge -> num];
587 }
588
589 int val_a, val_b, val_c;
590 for (j=1; j<i; j++)
591 {
592 val_a = tmp_ato[j];
593 val_b = tmp_atd[j];
594 val_c = tmp_fre[j];
595 for (k=j-1; k>-1; k--)
596 {
597 if (tmp_atd[k] < val_b) break;
598 tmp_ato[k+1] = tmp_ato[k];
599 tmp_atd[k+1] = tmp_atd[k];
600 tmp_fre[k+1] = tmp_fre[k];
601 }
602 tmp_ato[k+1] = val_a;
603 tmp_atd[k+1] = val_b;
604 tmp_fre[k+1] = val_c;
605 }
606 to_merge -> num = i;
607 g_free (to_merge -> list);
608 to_merge -> list = duplicate_int (to_merge -> num, tmp_ato);
609 g_free (tmp_ato);
610 g_free (to_merge -> list_id);
611 to_merge -> list_id = duplicate_int (to_merge -> num, tmp_atd);
612 g_free (tmp_atd);
613 g_free (to_merge -> frozen_id);
614 to_merge -> frozen_id = duplicate_int (to_merge -> num, tmp_fre);
615 g_free (tmp_fre);
616
617 if (upda)
618 {
619 for (i=0; i<to_merge -> num; i++)
620 {
621 j = to_merge -> list[i];
622 k = to_merge -> list_id[i];
623 for (l=0; l< tmp_fmol -> multi; l++)
624 {
625 if (tmp_proj -> atoms[0][j].coord[2] == tmp_fmol -> fragments[l]) break;
626 }
627 tmp_fmol -> atoms_id[k][l].a = to_merge -> id;
628 tmp_fmol -> atoms_id[k][l].b = i;
629 tmp_proj -> atoms[0][j].faid = to_merge -> id;
630 }
631 }

702 compare_non_bonded (to_merge -> name);
