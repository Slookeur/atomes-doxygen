173int open_save (FILE * fp, int i, int pid, int aid, int npi, gchar * pfile)

175 int j;

177
178 if (i == 0)
179 {
180 reading_input = TRUE;

182 reading_input = FALSE;
183 if (j != 0)
184 {
185 // error at read
186 if (pfile != NULL)
187 {
188 err = g_strdup_printf ("Impossible to open project file: \n%s\nError code: %d\n", pfile, j);
189 show_error (err, 0, MainWindow);

196#ifdef DEBUG

198#endif

200 add_project_to_workspace ();
201 prep_calc_actions ();
202 }
203 }
204 else
205 {
206 j = save_project (fp, get_project_by_id(pid), npi);
207 if (j != 0)
208 {
209 // error at write
210 if (pfile != NULL)
211 {
212 err = g_strdup_printf ("Impossible to save project file:\n%s\nError code: %d\n", pfile, j);
213 show_error (err, 0, MainWindow);
214 g_free (err);
215 }
216 }
217 else
218 {
219 if (pfile != NULL) get_project_by_id(pid) -> projfile = g_strdup_printf ("%s", pfile);
220 }
221 }
