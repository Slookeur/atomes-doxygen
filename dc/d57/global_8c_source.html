701gchar * calculation_time (gboolean modelv, double ctime)
702{
703 int i, j, k;
704 gchar * t_string = (modelv) ? "\n \tAnalysis was performed in: ": "";
705 if (ctime < 60.0)
706 {
707 return g_strdup_printf ("%s%f s", t_string, ctime);
708 }
709 else if (ctime < 3600.0)
710 {
711 i = ctime/60.0;
712 return g_strdup_printf ("%s%d m %f s", t_string, i, ctime-i*60.0);
713 }
714 else if (ctime < 86400.0)
715 {
716 // Really ?!
717 i = ctime/3600.0;
718 j = (ctime - i*3600.0)/60.0;
719 return g_strdup_printf ("%s%d h %d m %f s", t_string, i, j, ctime-i*3600.0-j*60.0);
720 }
721 else
722 {
723 // Seriously ?!
724 i = ctime/86400.0;
725 j = (ctime - i*86400.0)/3600.0;
726 k = (ctime - i*86400.0 - j*3600.0)/60.0;
727 return g_strdup_printf ("%s%d d %d h %d m %f s", t_string, i, j, k, ctime-i*86400.0-j*3600.0-k*60.0);
728 }
