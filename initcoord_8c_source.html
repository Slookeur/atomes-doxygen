71ColRGBA init_color (int id, int numid)
72{
73 float a, b;
74 ColRGBA col;
75 if (numid > 5)
76 {
77 a = (float)id;
78 b = (float)numid / 4.0;
79 if (a < b)
80 {
81 col.red = 1.0;
82 col.green = a / b;
83 col.blue = 0.0;
84 }
85 else if (a < 2*b)
86 {
87 col.red = 1.0 - (1.0 * (a-b)/b);
88 col.green = 1.0;
89 col.blue = 0.0;
90 }
91 else if (a < 3*b)
92 {
93 col.red = 0.0;
94 col.green = 1.0;
95 col.blue = (1.0 * (a-2.0*b)/b);
96 }
97 else
98 {
99 col.red = 0.0;
100 col.green = 1.0 - (1.0 * (a-3.0*b)/b);
101 col.blue = 1.0;
102 }
103 col.alpha = 1.0;
104 }
105 else
106 {
107 col = std[id];
108 col.alpha = 1.0;
109 }
110 return col;
111}
