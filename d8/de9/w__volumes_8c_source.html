144double get_sphere_caps_volume (double dab, double rad, double rbd)
145{
146 double h1, h2;
147 double px;
148 double rmi, rma;
149 rmi = min (rad, rbd);
150 rma = max (rad, rbd);
151 px = (dab*dab - rmi*rmi + rma*rma) / (2.0*dab);
152 h1 = rma - px;
153 h2 = rmi - (dab - px);
154 return cap_volume (h1, rma) + cap_volume (h2, rmi);
155}

164double sphere_volume (double rad)
165{
166 return ((4.0*pi)/3.0)*rad*rad*rad;

180double get_atoms_volume (struct project * this_proj, int rid, int sid, int gid, int gcid)
181{
182 int i, j, k, l, m, n, o;
183 double vol = 0.0;
184 double cap_vol = 0.0;
185 double * rvdws = allocdouble (this_proj -> nspec);
186 struct distance dist;
187 for (i=0; i<this_proj -> nspec; i++)
188 {
189 j = (int)this_proj -> chemistry -> chem_prop[CHEM_Z][i];
190 rvdws[i] = set_radius_ (& j, & rid);
191 }
192 for (j=0; j<2; j++)
193 {
194 for (k=0; k<this_proj -> modelgl -> bonds[sid][j]; k++)
195 {
196 l = this_proj -> modelgl -> bondid[sid][j][k][0];
197 m = this_proj -> atoms[sid][l].sp;
198 n = this_proj -> modelgl -> bondid[sid][j][k][1];
199 o = this_proj -> atoms[sid][n].sp;
200 if (gid < 0 || this_proj -> atoms[sid][l].coord[gid] == gcid)
201 {
202 dist = distance_3d (& this_proj -> cell, (this_proj -> cell.npt) ? sid : 0, & this_proj -> atoms[sid][l], & this_proj -> atoms[sid][n]);
203 if (dist.length < rvdws[m]+rvdws[o]) cap_vol += get_sphere_caps_volume (dist.length, rvdws[m], rvdws[o]);
204 }
205 }
206 }
207 vol = 0.0;
208 if (gid < 0)
209 {
210 for (i=0; i<this_proj -> nspec; i++)
211 {
212 vol += this_proj -> chemistry -> nsps[i] * sphere_volume (rvdws[i]);
213 }
214 vol -= cap_vol;
215 }
216 else
217 {
218 if (gid == 3)
219 {
220 for (i=0; i<this_proj -> nspec; i++)
221 {
222 vol += this_proj -> modelfc -> mols[sid][gcid].species[i] * sphere_volume (rvdws[i]);
223 }
224 vol *= this_proj -> modelfc -> mols[sid][gcid].multiplicity;
225 vol -= cap_vol;
226 vol /= this_proj -> modelfc -> mols[sid][gcid].multiplicity;
227 }
228 else
229 {
230 for (i=0; i<this_proj -> natomes; i++)
231 {
232 if (this_proj -> atoms[sid][i].coord[gid] == gcid)
233 {
234 j = this_proj -> atoms[sid][i].sp;
235 vol += sphere_volume (rvdws[j]);
236 }
237 }
238 vol -= cap_vol;
239 }
240 }
241 g_free (rvdws);
242 return vol;
243}

260double molecular_volume (int nats, struct atom * ats_vol, double baryc[3], double * rvdws, double a_ang, double b_ang, double c_ang)
261{
262 double paral[3][3];
263 mat4_t rot;
264 vec3_t c_old, c_new;
265 rot = m4_rotation_xyz (a_ang, b_ang, c_ang);
266 int i, j, k;
267 k = 0;
268 for (i=0; i<nats; i++)
269 {
270 j = ats_vol[i].sp;
271 c_old = vec3(ats_vol[i].x+baryc[0], ats_vol[i].y+baryc[1], ats_vol[i].z+baryc[2]);
272 c_new = m4_mul_pos (rot, c_old);
273 if (! k)
274 {
275 vamin[0] = c_new.x - rvdws[j];
276 vamax[0] = c_new.x + rvdws[j];
277 vamin[1] = c_new.y - rvdws[j];
278 vamax[1] = c_new.y + rvdws[j];
279 vamin[2] = c_new.z - rvdws[j];
280 vamax[2] = c_new.z + rvdws[j];
281 k ++;
282 }
283 else
284 {
285 vamin[0] = min(vamin[0], c_new.x - rvdws[j]);
286 vamax[0] = max(vamax[0], c_new.x + rvdws[j]);
287 vamin[1] = min(vamin[1], c_new.y - rvdws[j]);
288 vamax[1] = max(vamax[1], c_new.y + rvdws[j]);
289 vamin[2] = min(vamin[2], c_new.z - rvdws[j]);
290 vamax[2] = max(vamax[2], c_new.z + rvdws[j]);
291 }
292 }
293 paral[0][0] = vamax[0] - vamin[0];
294 paral[0][1] = 0.0;
295 paral[0][2] = 0.0;
296 paral[1][0] = 0.0;
297 paral[1][1] = vamax[1] - vamin[1];
298 paral[1][2] = 0.0;
299 paral[2][0] = 0.0;
300 paral[2][1] = 0.0;
301 paral[2][2] = vamax[2] - vamin[2];
302 ColRGBA null;
303 return draw_cuboid (FALSE, VOLMS, 0, m4_identity (), vec3(0.0,0.0,0.0), paral, null, 1.0);
304}

317double get_atoms_box (struct project * this_proj, int rid, int sid, int geo, int gid)
318{
319 int i, j;
320 double * rvdws = allocdouble (this_proj -> nspec);
321 for (i=0; i<this_proj -> nspec; i++)
322 {
323 j = (int)this_proj -> chemistry -> chem_prop[CHEM_Z][i];
324 rvdws[i] = set_radius_ (& j, & rid);
325 }
326 if (geo < 0) center_this_molecule (this_proj -> modelgl);
327 int a_ang, b_ang, c_ang;
328 double val, vbl;
329 int finess = pow (10, this_proj -> modelgl -> volume_win -> angp);
330 int nats;
331 struct atom * ats_vol = NULL;
332 struct insert_object * object = NULL;
333 gboolean rtmp;
334 double * baryc;
335 switch (geo)
336 {
337 case 2:
338 object_motion = TRUE;
339 rtmp = this_proj -> modelgl -> rebuild[0][0];
340 this_proj -> modelgl -> rebuild[0][0] = TRUE;
341 object = create_object_from_frag_mol (this_proj, geo, gid, NULL);
342 object_motion = FALSE;
343 this_proj -> modelgl -> rebuild[0][0] = rtmp;
344 nats = object -> atoms;
345 baryc = duplicate_double (3, object -> baryc);
346 ats_vol = object -> at_list;
347 break;
348 default:
349 nats = this_proj -> natomes;
350 ats_vol = this_proj -> atoms[sid];
351 baryc = allocdouble (3);
352 break;
353 }
354
355 for (a_ang=0; a_ang < 90*finess; a_ang ++)
356 {
357 for (b_ang=0; b_ang < 90*finess; b_ang ++)
358 {
359 for (c_ang=0; c_ang < 90*finess; c_ang ++)
360 {
361 val = molecular_volume (nats, ats_vol, baryc, rvdws, a_ang/finess, b_ang/finess, c_ang/finess);
362 if (a_ang == 0 && b_ang == 0 && c_ang == 0)
363 {
364 vbl = val;
365 if (geo < 0)
366 {
367 for (i=0; i<3; i++)
368 {
369 this_proj -> modelgl -> volume_box[rid][sid][i] = vamax[i] - vamin[i];
370 this_proj -> modelgl -> volume_box[rid][sid][i+6] = (vamax[i] + vamin[i]) / 2.0;
371 }
372 this_proj -> modelgl -> volume_box[rid][sid][3] = a_ang/finess;
373 this_proj -> modelgl -> volume_box[rid][sid][4] = b_ang/finess;
374 this_proj -> modelgl -> volume_box[rid][sid][5] = c_ang/finess;
375 }
376 else
377 {
378 for (i=0; i<3; i++)
379 {
380 this_proj -> modelgl -> frag_box[rid][sid][gid][i] = vamax[i] - vamin[i];
381 this_proj -> modelgl -> frag_box[rid][sid][gid][i+6] = (vamax[i] + vamin[i]) / 2.0;
382 }
383 this_proj -> modelgl -> frag_box[rid][sid][gid][3] = a_ang/finess;
384 this_proj -> modelgl -> frag_box[rid][sid][gid][4] = b_ang/finess;
385 this_proj -> modelgl -> frag_box[rid][sid][gid][5] = c_ang/finess;
386 }
387 }
388 else if (val < vbl)
389 {
390 vbl = val;
391 if (geo < 0)
392 {
393 for (i=0; i<3; i++)
394 {
395 this_proj -> modelgl -> volume_box[rid][sid][i] = vamax[i] - vamin[i];
396 this_proj -> modelgl -> volume_box[rid][sid][i+6] = (vamax[i] + vamin[i]) / 2.0;
397 }
398 this_proj -> modelgl -> volume_box[rid][sid][3] = a_ang/finess;
399 this_proj -> modelgl -> volume_box[rid][sid][4] = b_ang/finess;
400 this_proj -> modelgl -> volume_box[rid][sid][5] = c_ang/finess;
401 }
402 else
403 {
404 for (i=0; i<3; i++)
405 {
406 this_proj -> modelgl -> frag_box[rid][sid][gid][i] = vamax[i] - vamin[i];
407 this_proj -> modelgl -> frag_box[rid][sid][gid][i+6] = (vamax[i] + vamin[i]) / 2.0;
408 }
409 this_proj -> modelgl -> frag_box[rid][sid][gid][3] = a_ang/finess;
410 this_proj -> modelgl -> frag_box[rid][sid][gid][4] = b_ang/finess;
411 this_proj -> modelgl -> frag_box[rid][sid][gid][5] = c_ang/finess;
412 }
413 }
414 else
415 {
416 break;
417 }
418 }
419 }
420 }
421 ats_vol = NULL;
422 g_free (rvdws);
423 g_free (baryc);
424 if (geo == 2) g_free (object);
425 return vbl;
426}
